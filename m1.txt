//password strength testing using python
import re

while True:
    p = input("Enter password (or exit): ")

    if p.lower() == "exit":
        break

    if len(p) < 8:
        print("Weak")
    elif not re.search(r'[A-Z]', p):
        print("Add uppercase letter")
    elif not re.search(r'[a-z]', p):
        print("Add lowercase letter")
    elif not re.search(r'[0-9]', p):
        print("Add number")
    elif not re.search(r'[!@#$%^&*]', p):
        print("Add special character")
    else:
        print("Strong Password")

//basic firewall using advanced security
Open Windows Defender Firewall with Advanced Security. 
Click on Inbound Rules. 
Click New Rule,on the right panel.
Select Custom and click Next.
Choose All Programs → Click Next. 
Protocol and Ports: Keep default → Click Next. 
Under Remote IP address, select These IP addresses. 
o Click Add and enter the IP you want to block (example: 1.2.3.4). o Click OK → Next.
Action: Select Block the connection → Next. 
Profile: Select Domain, Private, Public → Next.
Name the rule (example: Manual_Block_IP) → Finish.

import requests
import csv
import subprocess

# Download IP blocklist from Abuse.ch
url = "https://feodotracker.abuse.ch/downloads/ipblocklist.csv"
response = requests.get(url).text

# Delete old firewall rule
delete_rule = 'netsh advfirewall firewall delete rule name="BadIP"'
subprocess.run(["powershell", "-Command", delete_rule])

# Read CSV data
mycsv = csv.reader(
    filter(lambda x: not x.startswith("#"), response.splitlines())
)

# Add firewall rules
for row in mycsv:
    ip = row[1]

    if ip != "dst_ip":
        print("Blocking IP:", ip)

        add_rule = (
            f'netsh advfirewall firewall add rule '
            f'name="BadIP" dir=out action=block remoteip={ip}'
        )

        subprocess.run(["powershell", "-Command", add_rule])
